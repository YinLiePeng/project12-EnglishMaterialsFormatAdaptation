# 任务列表管理功能技术方案

## 📋 项目概述

**目标：** 为现有任务列表添加完整的管理功能，包括批量删除、高级筛选、排序和统计功能

**技术栈：**
- 前端：React 19 + TypeScript + Tailwind CSS + Zustand + React Query
- 后端：FastAPI + SQLAlchemy (异步) + PostgreSQL/SQLite
- 当前状态：基础列表展示、状态筛选、分页

### 1.2 状态管理方案（Zustand Store）

```typescript
// src/store/taskStore.ts
import { create } from 'zustand';
import type { TaskFilter, TaskSortOption } from '../types';

interface TaskStore {
  // 筛选条件
  filters: TaskFilter;
  setFilters: (filters: Partial<TaskFilter>) => void;
  resetFilters: () => void;

  // 排序选项
  sortBy: TaskSortOption;
  setSortBy: (sortBy: TaskSortOption) => void;

  // 批量选择
  selectedTaskIds: string[];
  toggleTaskSelection: (taskId: string) => void;
  clearSelection: () => void;
  selectAll: (taskIds: string[]) => void;

  // 统计数据
  statistics: TaskStatistics | null;
  setStatistics: (stats: TaskStatistics) => void;

  // UI状态
  isFilterDrawerOpen: boolean;
  setFilterDrawerOpen: (open: boolean) => void;
}

export const useTaskStore = create<TaskStore>((set) => ({
  // 初始状态
  filters: {
    status: [],
    dateRange: null,
    layoutMode: [],
    presetStyle: [],
    hasError: null,
    searchKeyword: '',
  },
  sortBy: {
    field: 'created_at',
    order: 'desc',
  },
  selectedTaskIds: [],
  statistics: null,
  isFilterDrawerOpen: false,

  // Actions
  setFilters: (newFilters) =>
    set((state) => ({
      filters: { ...state.filters, ...newFilters },
    })),

  resetFilters: () =>
    set({
      filters: {
        status: [],
        dateRange: null,
        layoutMode: [],
        presetStyle: [],
        hasError: null,
        searchKeyword: '',
      },
    }),

  setSortBy: (sortBy) => set({ sortBy }),

  toggleTaskSelection: (taskId) =>
    set((state) => ({
      selectedTaskIds: state.selectedTaskIds.includes(taskId)
        ? state.selectedTaskIds.filter((id) => id !== taskId)
        : [...state.selectedTaskIds, taskId],
    })),

  clearSelection: () => set({ selectedTaskIds: [] }),

  selectAll: (taskIds) => set({ selectedTaskIds: taskIds }),

  setStatistics: (statistics) => set({ statistics }),

  setFilterDrawerOpen: (isFilterDrawerOpen) => set({ isFilterDrawerOpen }),
}));
```

```python
@router.get("")
async def list_tasks(
    page: int = 1,
    page_size: int = 20,
    status: str = None,      # 逗号分隔: "completed,failed"
    start_date: str = None,
    end_date: str = None,
    layout_mode: str = None,
    preset_style: str = None,
    has_error: bool = None,
    search: str = None,
    sort_by: str = "created_at",
    sort_order: str = "desc",
    db: AsyncSession = Depends(get_db)
):
    """获取任务列表（支持高级筛选和排序）"""
    query = select(Task)

    # 1. 状态筛选（多选）
    if status:
        status_list = status.split(',')
        query = query.where(Task.status.in_(status_list))

    # 2. 日期范围筛选
    if start_date:
        start = datetime.fromisoformat(start_date)
        query = query.where(Task.created_at >= start)
    if end_date:
        end = datetime.fromisoformat(end_date)
        query = query.where(Task.created_at <= end)

    # 3. 排版模式筛选
    if layout_mode:
        mode_list = layout_mode.split(',')
        query = query.where(Task.layout_mode.in_(mode_list))

    # 4. 预设样式筛选
    if preset_style:
        style_list = preset_style.split(',')
        query = query.where(Task.preset_style.in_(style_list))

    # 5. 错误状态筛选
    if has_error is not None:
        if has_error:
            query = query.where(Task.status == 'failed')
        else:
            query = query.where(Task.status.in_(['completed', 'pending', 'processing']))

    # 6. 关键词搜索（文件名模糊匹配）
    if search:
        query = query.where(Task.input_filename.ilike(f'%{search}%'))

    # 7. 排序
    sort_column = getattr(Task, sort_by, Task.created_at)
    if sort_order == 'asc':
        query = query.order_by(sort_column.asc())
    else:
        query = query.order_by(sort_column.desc())

    # 8. 分页
    offset = (page - 1) * page_size
    query = query.offset(offset).limit(page_size)

    # 9. 执行查询
    result = await db.execute(query)
    tasks = result.scalars().all()

    # 10. 获取总数（需要单独查询）
    count_query = select(func.count(Task.id))
    # ... 应用相同的筛选条件 ...
    total = await db.scalar(count_query)

    return {
        "code": 0,
        "data": {
            "total": total,
            "page": page,
            "page_size": page_size,
            "tasks": [task_to_dict(t) for t in tasks]
        }
    }
```

### 2.3 统计数据API

**端点：** `GET /api/v1/tasks/statistics`

**查询参数（可选）：**

**响应：**


**实现要点：**
```python
@router.get("/statistics")
async def get_task_statistics(
    start_date: str = None,
    end_date: str = None,
    db: AsyncSession = Depends(get_db)
):
    """获取任务统计数据"""
    # 基础查询
    query = select(Task)

    # 应用日期范围筛选
    if start_date:
        query = query.where(Task.created_at >= datetime.fromisoformat(start_date))
    if end_date:
        query = query.where(Task.created_at <= datetime.fromisoformat(end_date))

    # 获取所有任务
    result = await db.execute(query)
    tasks = result.scalars().all()

    # 计算统计数据
    total = len(tasks)
    pending = sum(1 for t in tasks if t.status == 'pending')
    processing = sum(1 for t in tasks if t.status == 'processing')
    completed = sum(1 for t in tasks if t.status == 'completed')
    failed = sum(1 for t in tasks if t.status == 'failed')

    # 计算平均处理时间（仅已完成的）
    completed_with_time = [t for t in tasks if t.processing_time is not None]
    avg_processing_time = (
        sum(t.processing_time for t in completed_with_time) / len(completed_with_time)
        if completed_with_time else None
    )

    # 计算完成率
    completion_rate = (completed / total * 100) if total > 0 else 0

    return {
        "code": 0,
        "data": {
            "total": total,
            "pending": pending,
            "processing": processing,
            "completed": completed,
            "failed": failed,
            "avg_processing_time": round(avg_processing_time, 2) if avg_processing_time else None,
            "completion_rate": round(completion_rate, 2)
        }
    }
```

### 3.2 查询优化建议

1. **使用count()替代获取所有记录**
   ```python
   # 不推荐
   total = len(tasks)

   # 推荐
   total = await db.scalar(select(func.count(Task.id)))
   ```

2. **使用select()替代直接查询**
   ```python
   # 只查询需要的字段
   query = select(Task.task_id, Task.status, Task.input_filename)
   ```

3. **使用批量操作**
   ```python
   # 批量删除
   await db.execute(
       delete(Task).where(Task.task_id.in_(task_ids))
   )
   ```

### 6.1 批量删除实现

**前端实现：**
```typescript
// src/hooks/useTaskBatchDelete.ts
import { useMutation, useQueryClient } from '@tanstack/react-query';
import { batchDeleteTasks } from '../services/api';

export function useTaskBatchDelete() {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: async (taskIds: string[]) => {
      return await batchDeleteTasks({ task_ids: taskIds, delete_files: true });
    },
    onSuccess: (data) => {
      // 刷新列表
      queryClient.invalidateQueries(['tasks']);

      // 显示结果
      if (data.failed_count > 0) {
        alert(`成功删除${data.deleted_count}项，失败${data.failed_count}项`);
      } else {
        alert(`成功删除${data.deleted_count}项`);
      }
    },
    onError: (error) => {
      alert('批量删除失败：' + error.message);
    },
  });
}
```

**后端实现：**
```python
@router.delete("/batch")
async def batch_delete_tasks(
    request: BatchDeleteRequest,
    db: AsyncSession = Depends(get_db)
):
    """批量删除任务"""
    # 1. 查询任务
    result = await db.execute(
        select(Task).where(Task.task_id.in_(request.task_ids))
    )
    tasks = result.scalars().all()

    # 2. 验证状态
    deletable_tasks = [
        t for t in tasks
        if t.status in ['completed', 'failed']
    ]

    if len(deletable_tasks) != len(request.task_ids):
        raise HTTPException(
            status_code=400,
            detail={
                "code": 400,
                "message": "只能删除已完成或失败的任务"
            }
        )

    # 3. 删除文件
    deleted_ids = []
    failed_ids = []

    for task in deletable_tasks:
        try:
            if request.delete_files:
                # 删除输入文件
                if task.input_file_path and Path(task.input_file_path).exists():
                    Path(task.input_file_path).unlink()

                # 删除输出文件
                if task.output_file_path and Path(task.output_file_path).exists():
                    Path(task.output_file_path).unlink()

            # 删除数据库记录
            await db.delete(task)
            deleted_ids.append(task.task_id)

        except Exception as e:
            logger.error(f"删除任务失败: {task.task_id}, 错误: {str(e)}")
            failed_ids.append(task.task_id)

    # 4. 提交事务
    await db.commit()

    return {
        "code": 0,
        "message": "批量删除完成",
        "data": {
            "deleted_count": len(deleted_ids),
            "failed_count": len(failed_ids),
            "failed_ids": failed_ids
        }
    }
```
