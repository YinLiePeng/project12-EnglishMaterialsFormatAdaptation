## 一、项目概述

### 1.1 项目定位
面向全国卷高三/初三英语教师的轻量化网站工具，以「规则引擎为主，大模型为辅」为核心技术架构，一站式解决教学资料的智能清洗、格式迁移适配、标准化排版、分级内容纠错问题。

### 1.2 核心目标
- **格式处理**: 100%确定性输出，无随机偏差
- **内容纠错**: 分级处理，全程留痕
- **易用性**: 核心操作不超过4步
- **安全性**: 零误删、零误改风险

### 1.3 技术原则
- 规则引擎承担所有格式控制
- 大模型仅负责语义理解与结构化输出
- 所有修改全程留痕
- **V1无用户系统**: 任何人访问即可使用，无需登录注册
- 存储策略分级管理：
  - **原始资料**: 临时存储（24小时），处理完成后自动过期
  - **处理结果**: 临时存储（24小时），用户下载后可过期
  - **公共模板**: 持久化存储，所有人共享

**V1简化架构说明**：
- **无Nginx**：V1无用户系统，直接使用Uvicorn部署
- **无Redis**：使用Python内置内存缓存（dict + TTL）存储临时文件
- **无Celery**：使用FastAPI BackgroundTasks处理异步任务
- **无Prometheus/Grafana**：使用Python logging + 健康检查端点
- **单容器部署**：简化部署流程，降低运维成本

```python
from docx import Document
from docx.shared import Pt, Inches, Cm
from docx.enum.text import WD_ALIGN_PARAGRAPH

class 格式规则引擎:
    """格式规则引擎核心"""

    def __init__(self, 排版方案: 排版方案):
        self.方案 = 排版方案

    def 应用页面设置(self, doc: Document):
        """应用页面设置"""
        sections = doc.sections
        for section in sections:
            section.page_height = Cm(self.方案.页面.高度)
            section.page_width = Cm(self.方案.页面.宽度)
            section.top_margin = Cm(self.方案.页面.上边距)
            section.bottom_margin = Cm(self.方案.页面.下边距)
            section.left_margin = Cm(self.方案.页面.左边距)
            section.right_margin = Cm(self.方案.页面.右边距)

    def 应用字体格式(self, run, 字体格式: 字体格式):
        """应用字体格式"""
        run.font.name = 字体格式.名称
        run.font.size = Pt(字体格式.大小)
        run.font.bold = 字体格式.加粗
        run.font.italic = 字体格式.斜体
        run.font.underline = 字体格式.下划线
        from docx.shared import RGBColor
        r, g, b = self._hex_to_rgb(字体格式.颜色)
        run.font.color.rgb = RGBColor(r, g, b)

    def 应用段落格式(self, paragraph, 段落格式: 段落格式):
        """应用段落格式"""
        pf = paragraph.paragraph_format

        # 对齐方式
        align_map = {
            对齐方式.左对齐: WD_ALIGN_PARAGRAPH.LEFT,
            对齐方式.居中: WD_ALIGN_PARAGRAPH.CENTER,
            对齐方式.右对齐: WD_ALIGN_PARAGRAPH.RIGHT,
            对齐方式.两端对齐: WD_ALIGN_PARAGRAPH.JUSTIFY
        }
        paragraph.alignment = align_map[段落格式.对齐]

        # 行间距
        if 段落格式.行间距:
            pf.line_spacing = 段落格式.行间距

        # 段前段后距
        if 段落格式.段前距:
            pf.space_before = Pt(段落格式.段前距)
        if 段落格式.段后距:
            pf.space_after = Pt(段落格式.段后距)

        # 缩进
        if 段落格式.首行缩进:
            pf.first_line_indent = Cm(段落格式.首行缩进)
        if 段落格式.左缩进:
            pf.left_indent = Cm(段落格式.左缩进)
        if 段落格式.右缩进:
            pf.right_indent = Cm(段落格式.右缩进)

    def 应用样式(self, doc: Document, 内容: str, 样式名: str):
        """应用完整样式"""
        # 创建新段落
        para = doc.add_paragraph(内容)
        style_def = getattr(self.方案, 样式名)

        # 应用段落格式
        self.应用段落格式(para, style_def.段落)

        # 应用字体格式到所有run
        for run in para.runs:
            self.应用字体格式(run, style_def.字体)

    @staticmethod
    def _hex_to_rgb(hex_color: str) -> tuple:
        """十六进制颜色转RGB"""
        hex_color = hex_color.lstrip('#')
        return tuple(int(hex_color[i:i+2], 16) for i in (0, 2, 4))
```

**大模型客户端（含容错机制）**:

```python
import aiohttp
import asyncio
import time
from typing import List, Dict, Any, Optional
import json

class 大模型服务熔断异常(Exception):
    """大模型服务熔断异常"""
    pass

class 大模型客户端:
    """大模型API客户端（含容错机制）"""

    def __init__(self, api_key: str, base_url: str, model: str):
        self.api_key = api_key
        self.base_url = base_url
        self.model = model
        self.session = None
        
        # 容错机制配置
        self.失败计数 = 0
        self.熔断状态 = "关闭"  # 关闭/打开/半开
        self.熔断阈值 = 5
        self.熔断恢复时间 = 60  # 秒
        self.熔断开始时间 = 0
        self.最大重试次数 = 3
        self.请求超时 = 30  # 秒

    async def __aenter__(self):
        self.session = aiohttp.ClientSession()
        return self

    async def __aexit__(self, exc_type, exc_val, exc_tb):
        if self.session:
            await self.session.close()

    async def 调用(
        self,
        prompt: str,
        system_prompt: str = None,
        temperature: float = 0.3,
        max_tokens: int = 20000  # 限制单次调用Token数（约15页内容）
    ) -> str:
        """调用大模型API（含容错机制）

        Token限制说明：
        - 最大Token数：20000（约15页内容）
        - 超出限制需要分批调用
        - 依据：单次API调用安全，避免超时和成本过高
        
        容错机制：
        - 熔断器：连续失败5次后熔断60秒
        - 重试：指数退避重试3次
        - 超时：单次请求30秒超时
        """
        # 检查Token限制
        if max_tokens > 20000:
            raise ValueError("Token数超出限制（最大20000）")
        
        # 检查熔断状态
        if self.熔断状态 == "打开":
            if time.time() - self.熔断开始时间 > self.熔断恢复时间:
                self.熔断状态 = "半开"
            else:
                raise 大模型服务熔断异常("服务暂时不可用，请稍后重试")

        # 带重试和退避的调用
        last_exception = None
        for attempt in range(self.最大重试次数):
            try:
                return await self._实际调用(prompt, system_prompt, temperature, max_tokens)
            except Exception as e:
                last_exception = e
                self.失败计数 += 1
                
                # 检查是否需要熔断
                if self.失败计数 >= self.熔断阈值:
                    self.熔断状态 = "打开"
                    self.熔断开始时间 = time.time()
                
                # 指数退避重试
                if attempt < self.最大重试次数 - 1:
                    await asyncio.sleep(2 ** attempt)  # 1秒, 2秒, 4秒
                else:
                    raise

    async def _实际调用(self, prompt: str, system_prompt: str = None, temperature: float = 0.3, max_tokens: int = 20000) -> str:
        """实际调用大模型API"""
        headers = {
            "Authorization": f"Bearer {self.api_key}",
            "Content-Type": "application/json"
        }

        messages = []
        if system_prompt:
            messages.append({"role": "system", "content": system_prompt})
        messages.append({"role": "user", "content": prompt})

        payload = {
            "model": self.model,
            "messages": messages,
            "temperature": temperature,
            "max_tokens": max_tokens
        }

        timeout = aiohttp.ClientTimeout(total=self.请求超时)
        async with self.session.post(
            f"{self.base_url}/chat/completions",
            headers=headers,
            json=payload,
            timeout=timeout
        ) as response:
            response.raise_for_status()
            data = await response.json()
            
            # 成功调用，重置失败计数
            self.失败计数 = 0
            if self.熔断状态 == "半开":
                self.熔断状态 = "关闭"
            
            return data["choices"][0]["message"]["content"]

    async def 调用并解析(
        self,
        prompt: str,
        system_prompt: str = None,
        输出模型: BaseModel = 大模型输出
    ) -> BaseModel:
        """调用大模型并解析JSON输出"""
        # 第一轮调用
        response = await self.调用(prompt, system_prompt)

        # 尝试解析JSON
        try:
            # 提取JSON（处理可能的前后文本）
            json_str = self._提取_json(response)
            data = json.loads(json_str)
            return 输出模型(**data)
        except Exception as e:
            # 解析失败，要求重新输出
            retry_prompt = f"{prompt}\n\n你的输出格式有误，请严格按照JSON格式重新输出。"
            response = await self.调用(retry_prompt, system_prompt)
            json_str = self._提取_json(response)
            data = json.loads(json_str)
            return 输出模型(**data)

    async def 降级处理(self, prompt: str, 任务类型: str) -> "大模型输出":
        """服务不可用时的降级策略"""

        class 降级任务类型(Enum):

            内容纠错 = "correct"
            结构校验 = "structure"

            )

    @staticmethod
    def _提取_json(text: str) -> str:
        """从文本中提取JSON"""
        import re

        # 尝试直接匹配
        try:
            json.loads(text)
            return text
        except:
            pass

        # 尝试提取```json...```
        pattern = r'```json\s*(.*?)\s*```'
        match = re.search(pattern, text, re.DOTALL)
        if match:
            return match.group(1)

        # 尝试提取{...}
        pattern = r'\{.*\}'
        match = re.search(pattern, text, re.DOTALL)
        if match:
            return match.group(0)

        # 尝试提取[...]
        pattern = r'\[.*\]'
        match = re.search(pattern, text, re.DOTALL)
        if match:
            return match.group(0)

        raise ValueError("无法从输出中提取JSON")
```

```python
import fitz  # PyMuPDF
import pdfplumber
from pathlib import Path
from typing import List, Dict, Any, Tuple
import re

class PDF处理器:
    """PDF文件处理器"""

    def __init__(self, file_path: Path):
        self.file_path = file_path
        self.doc = fitz.open(file_path)

    def 提取原生内容(self) -> List[Dict[str, Any]]:
        """提取原生PDF内容（支持分栏检测）"""
        页面内容列表 = []

        for page_num, page in enumerate(self.doc):
            # 检测分栏
            栏列表 = self._检测分栏(page)

            if len(栏列表) > 1:
                # 多栏页面
                页面内容 = {
                    "页码": page_num + 1,
                    "类型": "多栏",
                    "栏数": len(栏列表),
                    "内容块": []
                }

                for 栏索引, 栏区域 in enumerate(栏列表):
                    栏内容 = page.get_text("text", clip=栏区域)
                    页面内容["内容块"].append({
                        "栏索引": 栏索引,
                        "区域": 栏区域,
                        "文本": 栏内容
                    })

                页面内容列表.append(页面内容)
            else:
                # 单栏页面
                文本 = page.get_text("text")
                页面内容 = {
                    "页码": page_num + 1,
                    "类型": "单栏",
                    "文本": 文本
                }
                页面内容列表.append(页面内容)

        return 页面内容列表

    def 提取表格(self, page_num: int) -> List[Dict[str, Any]]:
        """提取指定页的表格"""
        if page_num >= len(self.doc):
            return []

        page = self.doc[page_num]
        表格列表 = page.find_tables()

        结果 = []
        for table in 表格列表:
            表格数据 = {
                "表格索引": len(结果),
                "位置": table.bbox,
                "行数": table.row_count,
                "列数": table.col_count,
                "数据": table.extract()
            }
            结果.append(表格数据)

        return 结果

    def 提取图片(self, page_num: int) -> List[Dict[str, Any]]:
        """提取指定页的图片"""
        if page_num >= len(self.doc):
            return []

        page = self.doc[page_num]
        图片列表 = page.get_images()

        结果 = []
        for img_index, img in enumerate(图片列表):
            xref = img[0]
            base_image = self.doc.extract_image(xref)

            图片数据 = {
                "图片索引": img_index,
                "XREF": xref,
                "格式": base_image["ext"],
                "数据": base_image["image"],
                "位置": page.get_image_bbox(img)
            }
            结果.append(图片数据)

        return 结果

    def 合并跨页内容(self, 内容列表: List[Dict[str, Any]]) -> List[str]:
        """合并跨页断句的内容"""
        合并后内容 = []

        for 页面内容 in 内容列表:
            if 页面内容["类型"] == "单栏":
                合并后内容.append(页面内容["文本"])
            else:
                # 多栏页面，需要保持顺序
                for 内容块 in 页面内容["内容块"]:
                    合并后内容.append(内容块["文本"])

        # 处理跨页断句
        # 例如：页面末尾的单词被拆分 "comput-" 和 "er"
        最终内容 = []
        for i, 段落 in enumerate(合并后内容):
            if i > 0:
                上一段 = 最终内容[-1]
                # 检查末尾是否有连字符
                if 上一段.endswith('-'):
                    # 去掉连字符，合并当前段落开头的单词
                    最终内容[-1] = 上一段[:-1] + 段落
                else:
                    最终内容.append(段落)
            else:
                最终内容.append(段落)

        return 最终内容

    def 转换为结构化内容(self) -> List[Dict[str, Any]]:
        """将PDF内容转换为结构化格式（类似DOCX）"""
        页面内容 = self.提取原生内容()
        结构化内容 = []

        for 页面 in 页面内容:
            if 页面["类型"] == "单栏":
                paragraphs = self._分割段落(页面["文本"])
                for para_text in paragraphs:
                    if para_text.strip():
                        结构化内容.append({
                            "类型": "段落",
                            "文本": para_text,
                            "页码": 页面["页码"]
                        })
            else:
                for 内容块 in 页面["内容块"]:
                    paragraphs = self._分割段落(内容块["文本"])
                    for para_text in paragraphs:
                        if para_text.strip():
                            结构化内容.append({
                                "类型": "段落",
                                "文本": para_text,
                                "页码": 页面["页码"],
                                "栏索引": 内容块["栏索引"]
                            })
```

```python
from docx import Document
from docx.shared import Pt, RGBColor
from docx.enum.text import WD_ALIGN_PARAGRAPH
from typing import List, Dict, Any, Optional
import fitz  # PyMuPDF

class PDF样式提取器:
    """从PDF中提取文本样式信息"""

    def 提取文本样式(self, doc: fitz.Document, page_num: int) -> List[Dict]:
        """提取PDF文本的样式信息"""
        page = doc[page_num]
        样式列表 = []

        # 使用PyMuPDF提取文本块的样式信息
        blocks = page.get_text("dict")
        for block in blocks["blocks"]:
            if block["type"] == 0:  # 文本块
                for line in block["lines"]:
                    for span in line["spans"]:
                        样式列表.append({
                            "文本": span["text"],
                            "字体": span["font"],
                            "字号": span["size"],
                            "颜色": span["color"],
                            "加粗": bool(span["flags"] & 2**4),
                            "斜体": bool(span["flags"] & 2**1),
                            "位置": span["origin"]
                        })
        return 样式列表

class PDF样式映射器:
    """PDF样式到DOCX样式的映射器"""

    # 字体映射表
    字体映射 = {
        "TimesNewRomanPSMT": "Times New Roman",
        "TimesNewRoman": "Times New Roman",
        "ArialMT": "Arial",
        "Arial": "Arial",
        "SimSun": "宋体",
        "SimHei": "黑体",
        "KaiTi": "楷体",

        "MicrosoftYaHei": "微软雅黑"
    }

    def 映射段落样式(self, PDF样式: Dict) -> Dict:
        """将PDF样式映射为DOCX段落样式"""
        return {
            "字体": self.字体映射.get(PDF样式["字体"], "宋体"),
            "字号": self._映射字号(PDF样式["字号"]),
            "加粗": PDF样式.get("加粗", False),
            "斜体": PDF样式.get("斜体", False),
            "颜色": self._映射颜色(PDF样式.get("颜色", 0))
        }

    def _映射字号(self, PDF字号: float) -> float:
        """映射PDF字号到DOCX字号"""
        # PDF字号通常是磅值，直接使用
        return round(PDF字号, 1)

    def _映射颜色(self, PDF颜色: int) -> str:
        """映射PDF颜色到DOCX颜色（十六进制）"""
        if PDF颜色 == 0:
            return "000000"  # 黑色
        return f"{PDF颜色:06x}"

class PDF转DOCX生成器:
    """PDF内容转DOCX文档生成器"""

    def __init__(self):
        self.doc = Document()
        self.样式映射器 = PDF样式映射器()
        self.样式提取器 = PDF样式提取器()

    def 生成文档(
        self,
        内容列表: List[Dict],
        样式列表: List[Dict],
        表格列表: List[Dict] = None,
        图片列表: List[Dict] = None
    ) -> Document:
        """生成格式化DOCX文档

        Args:
            内容列表: 结构化内容列表（段落、表格、图片）
            样式列表: PDF样式信息列表
            表格列表: 表格数据列表
            图片列表: 图片数据列表

        Returns:
            生成的DOCX文档对象
        """
        # 1. 插入文本内容
        for 内容项 in 内容列表:
            if 内容项["类型"] == "段落":
                self._插入段落(内容项, 样式列表)
            elif 内容项["类型"] == "表格":
                self._插入表格(内容项)
            elif 内容项["类型"] == "图片":
                self._插入图片(内容项)

        # 2. 插入额外的表格
        if 表格列表:
            for 表格 in 表格列表:
                self._插入表格数据(表格)

        # 3. 插入额外的图片
        if 图片列表:
            for 图片 in 图片列表:
                self._插入图片数据(图片)

        return self.doc

    def _插入段落(self, 内容项: Dict, 样式列表: List[Dict]):
        """插入段落并应用样式"""
        文本 = 内容项["文本"]
        if not 文本.strip():
            return

        paragraph = self.doc.add_paragraph(文本)

        # 查找对应的样式
        对应样式 = self._查找对应样式(文本, 样式列表)
        if 对应样式:
            self._应用段落样式(paragraph, 对应样式)

    def _应用段落样式(self, paragraph, 样式: Dict):
        """应用段落样式"""
        if not paragraph.runs:
            return

        run = paragraph.runs[0]

        # 应用字体样式
        映射样式 = self.样式映射器.映射段落样式(样式)
        run.font.name = 映射样式["字体"]
        run.font.size = Pt(映射样式["字号"])
        run.font.bold = 映射样式["加粗"]
        run.font.italic = 映射样式["斜体"]

        # 应用颜色
        if 映射样式["颜色"]:
            try:
                r, g, b = self._hex_to_rgb(映射样式["颜色"])
                run.font.color.rgb = RGBColor(r, g, b)
            except:
                pass  # 颜色转换失败时忽略

    def _插入表格(self, 内容项: Dict):
        """插入表格"""
        表格数据 = 内容项.get("数据", [])
        if not 表格数据:
            return

        行数 = len(表格数据)
        列数 = max(len(row) for row in 表格数据) if 表格数据 else 0

        if 行数 == 0 or 列数 == 0:
            return

        table = self.doc.add_table(rows=行数, cols=列数)
        table.style = 'Table Grid'

        for i, row in enumerate(表格数据):
            for j, cell_text in enumerate(row):
                if j < 列数:
                    table.cell(i, j).text = str(cell_text) if cell_text else ""

    def _插入图片(self, 内容项: Dict):
        """插入图片"""
        图片数据 = 内容项.get("数据")
        if not 图片数据:
            return

        from io import BytesIO
        try:
            图片流 = BytesIO(图片数据)
            self.doc.add_picture(图片流)
        except Exception as e:
            # 图片插入失败时添加占位文本
            self.doc.add_paragraph(f"[图片插入失败: {str(e)}]")

    def _插入表格数据(self, 表格: Dict):
        """插入表格数据"""
        表格数据 = 表格.get("数据", [])
        if 表格数据:
            内容项 = {"类型": "表格", "数据": 表格数据}
            self._插入表格(内容项)

    def _插入图片数据(self, 图片: Dict):
        """插入图片数据"""
        图片数据 = 图片.get("数据")
        if 图片数据:
            内容项 = {"类型": "图片", "数据": 图片数据}
            self._插入图片(内容项)

    def _查找对应样式(self, 文本: str, 样式列表: List[Dict]) -> Optional[Dict]:
        """在样式列表中查找对应的样式"""
        if not 样式列表:
            return None

        # 精确匹配
        for 样式 in 样式列表:
            if 样式["文本"] and 文本.startswith(样式["文本"][:20]):
                return 样式

        # 模糊匹配（前10个字符）
        for 样式 in 样式列表:
            if 样式["文本"] and 文本[:10] in 样式["文本"]:
                return 样式

        return None

    def _hex_to_rgb(self, hex_color: str) -> tuple:
        """十六进制颜色转RGB"""
        hex_color = hex_color.lstrip('#')
        if len(hex_color) == 6:
            return tuple(int(hex_color[i:i+2], 16) for i in (0, 2, 4))
        return (0, 0, 0)  # 默认黑色

    def 保存(self, output_path: str):
        """保存文档"""
        self.doc.save(output_path)
```

```python
from typing import List, Dict, Any, Set
```

```python
    选项模式 = [
        r'^[A-D][\.\)]',  # A. B) C. D)
        r'^\([A-D]\)',    # (A) (B) (C) (D)
    ]
    
    # 改错题关键词
    改错题关键词 = [
        "改错", "error correction", "找出错误", "correct the errors",
        "correct the mistakes", "每行一处错误"
    ]

    def 识别核心区域(self, 内容结构: List[Dict]) -> List[Dict]:
        """识别核心考察内容区域并标记"""
        import re
        
        标记后内容 = []
        当前区域类型 = 核心区域类型.非核心区域
        当前是否核心区域 = False
        
        for 内容项 in 内容结构:
            文本 = 内容项.get("文本", "")
            类型 = 内容项.get("类型", "")
            
            # 检测题号 → 进入题干区域
            if self._是题号(文本):
                当前区域类型 = 核心区域类型.题干
                当前是否核心区域 = True
            
            # 检测选项 → 选项区域
            elif self._是选项(文本) and 当前是否核心区域:
                当前区域类型 = 核心区域类型.选项
            
            # 检测改错题
            elif self._是改错题内容(文本):
                当前区域类型 = 核心区域类型.改错题原文
                当前是否核心区域 = True
            
            # 检测完形填空
            elif self._是完形填空(文本):
                当前区域类型 = 核心区域类型.完形填空原文
                当前是否核心区域 = True
            
            # 添加区域标记
            内容项["是否核心区域"] = 当前是否核心区域
            内容项["核心区域类型"] = 当前区域类型.value
            标记后内容.append(内容项)
        
        return 标记后内容

    def 检查修改权限(self, 错误项: 修改项, 内容结构: List[Dict]) -> bool:
        """检查修改是否被允许"""
        # 找到错误项所在的内容项
        所在内容项 = self._查找所在内容项(错误项.位置, 内容结构)
        
        if 所在内容项 and 所在内容项.get("是否核心区域"):
            # 核心区域内，检查修改类型
            if 错误项.类型 in self.核心区域禁止修改:
                return False  # 禁止修改
        
        return True

    def 获取修改建议(self, 错误项: 修改项, 内容结构: List[Dict]) -> str:
        """获取修改建议（核心区域的特殊提示）"""
        所在内容项 = self._查找所在内容项(错误项.位置, 内容结构)
        
        if 所在内容项 and 所在内容项.get("是否核心区域"):
            区域类型 = 所在内容项.get("核心区域类型")
            
            if 区域类型 == 核心区域类型.改错题原文.value:
                return "【改错题】此错误可能是故意设置的，已标注但未自动修正"
            elif 错误项.类型 in self.核心区域禁止修改:
                return "【核心区域】此区域仅允许修正标点/空格/换行，已标注但未自动修正"
        
        return 错误项.原因

    def _是题号(self, 文本: str) -> bool:
        """检测是否为题号"""
        import re
        for pattern in self.题号模式:
            if re.match(pattern, 文本.strip()):
                return True
        return False

    def _是选项(self, 文本: str) -> bool:
        """检测是否为选项"""
        import re
        for pattern in self.选项模式:
            if re.match(pattern, 文本.strip()):
                return True
        return False

    def _是改错题内容(self, 文本: str) -> bool:
        """检测是否为改错题内容"""
        文本小写 = 文本.lower()
        return any(keyword in 文本小写 for keyword in self.改错题关键词)

    def _是完形填空(self, 文本: str) -> bool:
        """检测是否为完形填空"""
        import re
        # 完形填空特征：包含多个下划线或空格占位符
        下划线数量 = len(re.findall(r'_{2,}', 文本))
        return 下划线数量 >= 3

    def _查找所在内容项(self, 位置: 位置信息, 内容结构: List[Dict]) -> Optional[Dict]:
        """根据位置信息查找所在的内容项"""
        for 内容项 in 内容结构:
            if 内容项.get("段落索引") == 位置.段落索引:
                return 内容项
        return None
```

```python
                修正后列表.append(错误项)

            return False

        # 规则3：只允许4类无歧义错误
        # 这里需要更复杂的拼写错误检测逻辑
        # 例如：编辑距离算法

        return True

            if 错误项.操作 == 操作类型.替换:

                    "原文": 错误项.原文,
                    "替换文本": 错误项.替换内容,
                    "原因": 错误项.原因
                })
            elif 错误项.操作 == 操作类型.标注:
                # 仅标注：添加批注
                修订列表.append({
                    "类型": "批注",
                    "位置": 错误项.位置,
                    "原文": 错误项.原文,
                    "批注文本": f"疑似问题：{错误项.原因}",
                    "原因": 错误项.原因
                })

        return 修订列表

    def _加载词典(self, 词典路径: str) -> Set[str]:
        """加载词典文件"""
        path = Path(词典路径)

        if not path.exists():
            # 如果词典文件不存在，返回空集合
            return set()

        with open(path, 'r', encoding='utf-8') as f:
            data = json.load(f)
            return set(data.get("words", []))
```

**实现方式**: 使用python-docx提取模板的样式定义

```python
from docx import Document
from typing import Dict, List, Optional

@dataclass
class 样式信息:
    """样式详细信息"""
    名称: str
    样式ID: str
    基础样式: Optional[str]
    字体: dict
    段落格式: dict
    层级: int
    是否有效: bool  # 样式层级是否有效

class 样式层级提取器:

    def 提取样式层级体系(self) -> Dict[str, 样式信息]:

                样式信息 = self._分析样式(style)
                self.样式库[style.name] = 样式信息

        # 检测样式层级是否混乱
        self.检测样式层级有效性()

        return self.样式库

    def _分析样式(self, style) -> 样式信息:
        """分析单个样式"""
        return 样式信息(
            名称=style.name,
            样式ID=style.style_id,
            基础样式=style.base_style.name if style.base_style else None,
            字体=self._提取字体信息(style),
            段落格式=self._提取段落格式(style),
            层级=self._计算样式层级(style),
            是否有效=True  # 默认有效，后续检测
        )

    def _计算样式层级(self, style) -> int:
        """计算样式的层级（基于继承关系）"""
        层级 = 0
        当前样式 = style
        已访问 = set()

        while 当前样式.base_style:
            if 当前样式.name in 已访问:
                break  # 防止循环继承
            已访问.add(当前样式.name)
            层级 += 1
            当前样式 = 当前样式.base_style
            if 层级 > 10:  # 防止过深继承
                break

        return 层级

    def 检测样式层级有效性(self) -> bool:
        """检测样式层级是否混乱"""
        混乱标记 = False

        for 名称, 样式 in self.样式库.items():
            # 检测1：基础样式不存在
            if 样式.基础样式 and 样式.基础样式 not in self.样式库:
                样式.是否有效 = False
                混乱标记 = True

            # 检测2：层级过深（超过5层）
            if 样式.层级 > 5:
                样式.是否有效 = False
                混乱标记 = True

            # 检测3：样式名称为空或异常
            if not 样式.名称 or len(样式.名称) > 100:
                样式.是否有效 = False
                混乱标记 = True

        return 混乱标记

    def 获取样式层级树(self) -> List[Dict]:
        """获取样式层级树（用于前端展示）"""
        层级树 = []

        # 按层级分组
        按层级分组 = {}
        for 名称, 样式 in self.样式库.items():
            层级 = 样式.层级
            if 层级 not in 按层级分组:
                按层级分组[层级] = []
            按层级分组[层级].append({
                "名称": 名称,
                "样式ID": 样式.样式ID,
                "字体": 样式.字体,
                "是否有效": 样式.是否有效
            })

        # 构建树形结构
        for 层级 in sorted(按层级分组.keys()):
            层级树.append({
                "层级": 层级,
                "样式列表": 按层级分组[层级]
            })

        return 层级树
```

#### 4.8.3 内容结构识别

**实现方式**: 规则引擎为主 + 大模型为辅

```python
class 内容结构识别器:
    """识别原始资料的内容结构"""

    # 内容类型枚举
    内容类型 = {
        "主标题": "主标题",
        "子标题": "子标题",
        "题号": "题号",
        "选项": "选项",
        "正文": "正文",
        "答案": "答案",
        "解析": "解析"
    }

    def 识别内容结构(self, 内容列表: List[Dict]) -> List[Dict]:
        """使用规则引擎识别内容结构"""
        识别结果 = []

        for 内容项 in 内容列表:
            文本 = 内容项.get("文本", "")
            字体大小 = 内容项.get("字体大小", 12)
            对齐方式 = 内容项.get("对齐方式", "左对齐")
            是否加粗 = 内容项.get("是否加粗", False)

            # 规则识别
            类型, 置信度 = self._规则识别(文本, 字体大小, 对齐方式, 是否加粗)

            识别结果.append({
                **内容项,
                "类型": 类型,
                "置信度": 置信度
            })

        return 识别结果

    def _规则识别(self, 文本: str, 字体大小: float, 对齐方式: str, 是否加粗: bool) -> tuple:
        """规则识别单个内容项"""
        # 规则1：字号最大且居中 → 主标题
        if 字体大小 >= 18 and 对齐方式 == "居中":
            return "主标题", 0.9

        # 规则2：字号较大且居中 → 子标题
        if 字体大小 >= 14 and 对齐方式 == "居中":
            return "子标题", 0.85

        # 规则3：字号较大且加粗 → 子标题
        if 字体大小 >= 14 and 是否加粗:
            return "子标题", 0.8

        # 规则4：匹配题号格式
        if self._匹配题号格式(文本):
            return "题号", 0.95

        # 规则5：匹配选项格式
        if self._匹配选项格式(文本):
            return "选项", 0.9

        # 规则6：其他 → 正文
        return "正文", 0.7

    def _匹配题号格式(self, 文本: str) -> bool:
        """匹配题号格式"""
        import re
        patterns = [
            r'^\d+[\.\、]',           # 1. / 1、
            r'^[\(（]\d+[）\)]',      # (1) / （1）
            r'^第[一二三四五六七八九十]+[题节]',  # 第一题
        ]
        return any(re.match(p, 文本) for p in patterns)

    def _匹配选项格式(self, 文本: str) -> bool:
        """匹配选项格式"""
        import re
        patterns = [
            r'^[A-D][\.\、]',         # A. / A、
            r'^[\(（][A-D][）\)]',    # (A) / （A）
        ]
        return any(re.match(p, 文本) for p in patterns)
```

```python
class 百度OCR客户端:

        self.secret_key = secret_key
```

### 7.3 文件安全

**文件上传安全**:
```python
import magic
import hashlib
from pathlib import Path
from typing import Tuple

ALLOWED_EXTENSIONS = {".docx", ".pdf"}
ALLOWED_MIME_TYPES = {
    "application/vnd.openxmlformats-officedocument.wordprocessingml.document",
    "application/pdf"
}
MAX_FILE_SIZE = 50 * 1024 * 1024  # 50MB
DOCX_MAX_PAGES = 30             # DOCX最大页数（超过30页修订可能导致Word卡顿）
PDF_MAX_PAGES = 50              # PDF最大页数（OCR成本和时间可控）
MAX_REVISIONS = 500             # 最大修订数量（超出可能使Word打开变慢）

async def 验证上传文件(
    file_content: bytes,
    filename: str
) -> Tuple[bool, str]:
    """验证上传文件的安全性"""

    # 1. 文件大小检查
    if len(file_content) > MAX_FILE_SIZE:
        return False, "文件大小超过限制（最大50MB）"

    # 2. 文件扩展名检查
    file_ext = Path(filename).suffix.lower()
    if file_ext not in ALLOWED_EXTENSIONS:
        return False, f"不支持的文件格式：{file_ext}"

    # 3. MIME类型检查
    mime = magic.from_buffer(file_content, mime=True)
    if mime not in ALLOWED_MIME_TYPES:
        return False, f"文件类型不匹配：{mime}"

    # 4. 页数检查（异步调用）
    if file_ext == ".docx":
        page_count = await 检查DOCX页数(file_content)
        if page_count > DOCX_MAX_PAGES:
            return False, f"DOCX页数超过限制（最大{DOCX_MAX_PAGES}页，当前{page_count}页）"
    elif file_ext == ".pdf":
        page_count = await 检查PDF页数(file_content)
        if page_count > PDF_MAX_PAGES:
            return False, f"PDF页数超过限制（最大{PDF_MAX_PAGES}页，当前{page_count}页）"

    return True, ""

async def 检查DOCX页数(file_content: bytes) -> int:
    """检查DOCX文件页数"""
    from docx import Document
    import io

    doc = Document(io.BytesIO(file_content))
    # 估算页数（基于段落数，每页约30-40段）
    # 更精确的方法是解析XML中的分页符
    段落数 = len(doc.paragraphs)
    估算页数 = max(1, 段落数 // 35)
    return 估算页数

async def 检查PDF页数(file_content: bytes) -> int:
    """检查PDF文件页数"""
    import fitz
    import io

    doc = fitz.open(stream=file_content, filetype="pdf")
    页数 = len(doc)
    doc.close()
    return 页数

async def 检查修订数量(文档内容) -> int:
    """检查文档中的修订数量"""
    # 使用docx-revisions库检查修订数量
    from docx_revisions import RevisionDocument

    rdoc = RevisionDocument(文档内容)
    修订数量 = sum(1 for _ in rdoc.iter_revisions())
    return 修订数量

def 计算文件哈希(file_content: bytes) -> str:
    """计算文件SHA256哈希"""
    return hashlib.sha256(file_content).hexdigest()

def 生成安全文件名(original_filename: str) -> str:
    """生成安全的文件名"""
    import uuid
    ext = Path(original_filename).suffix
    return f"{uuid.uuid4()}{ext}"
```

```python
# 文件不持久化原则
class 文件处理器:
    def __init__(self):
        self.临时存储 = {}  # 内存存储
        self.过期时间 = {}  # 过期时间

    async def 保存临时文件(self, file_id: str, content: bytes):
        """保存到内存，设置24小时过期"""
        self.临时存储[file_id] = content
        self.过期时间[file_id] = time.time() + 86400

    async def 获取临时文件(self, file_id: str) -> Optional[bytes]:
        """获取临时文件"""
        if file_id not in self.临时存储:
            return None

        # 检查是否过期
        if time.time() > self.过期时间.get(file_id, 0):
            del self.临时存储[file_id]
            del self.过期时间[file_id]
            return None

        return self.临时存储[file_id]

    async def 清理过期文件(self):
        """定期清理过期文件"""
        当前时间 = time.time()
        过期文件 = [
            file_id for file_id, expire_time in self.过期时间.items()
            if 当前时间 > expire_time
        ]

        for file_id in 过期文件:
            del self.临时存储[file_id]
            del self.过期时间[file_id]
```

```python
from fastapi import BackgroundTasks
from typing import Dict, Any
import asyncio

class 异步任务服务:
    """使用FastAPI BackgroundTasks处理异步任务（V1简化版）"""

    def __init__(self):
        self.任务状态: Dict[str, Dict[str, Any]] = {}

    def 提交任务(self, task_id: str, background_tasks: BackgroundTasks, 处理参数: dict):
        """提交异步任务"""
        self.任务状态[task_id] = {"status": "pending"}
        background_tasks.add_task(self._执行任务, task_id, 处理参数)

    async def _执行任务(self, task_id: str, 处理参数: dict):
        """执行异步任务"""
        try:
            self.任务状态[task_id]["status"] = "processing"
            
            # 执行处理流水线
            结果 = await 执行处理流水线(处理参数)
            
            self.任务状态[task_id] = {
                "status": "completed",
                "result": 结果
            }
        except Exception as e:
            self.任务状态[task_id] = {
                "status": "failed",
                "error": str(e)
            }

    def 获取任务状态(self, task_id: str) -> Dict[str, Any]:
        """获取任务状态"""
        return self.任务状态.get(task_id, {"status": "not_found"})
```

### 8.2 缓存策略

**内存缓存（dict + TTL）**:
```python
from typing import Optional, Any, Dict
import time
import json

class 内存缓存服务:
    """使用Python内存缓存（V1简化版）"""

    def __init__(self, default_ttl: int = 3600):
        self.缓存: Dict[str, Dict[str, Any]] = {}
        self.default_ttl = default_ttl

    def 获取(self, key: str) -> Optional[Any]:
        """获取缓存"""
        if key in self.缓存:
            item = self.缓存[key]
            if time.time() < item["expire_at"]:
                return item["value"]
            else:
                # 过期，删除
                del self.缓存[key]
        return None

    def 设置(self, key: str, value: Any, expire_seconds: int = None):
        """设置缓存"""
        if expire_seconds is None:
            expire_seconds = self.default_ttl
        self.缓存[key] = {
            "value": value,
            "expire_at": time.time() + expire_seconds
        }

    def 删除(self, key: str):
        """删除缓存"""
        if key in self.缓存:
            del self.缓存[key]

    def 清理过期(self):
        """清理过期缓存"""
        now = time.time()
        expired_keys = [k for k, v in self.缓存.items() if now >= v["expire_at"]]
        for key in expired_keys:
            del self.缓存[key]

    # 特定缓存方法
    def 缓存任务状态(self, task_id: str, task_data: dict):
        key = f"task:{task_id}"
        self.设置(key, task_data, expire_seconds=86400)  # 24小时

    def 获取任务状态(self, task_id: str) -> Optional[dict]:
        key = f"task:{task_id}"
        return self.获取(key)

# 全局缓存实例
缓存实例 = 内存缓存服务()
```

```python
async def 获取用户任务列表(user_id: int, limit: int = 10):
    async with AsyncSessionLocal() as session:
        # 使用索引字段user_id

            .order_by(Task.created_at.desc())
            .limit(limit)
        )

    async with AsyncSessionLocal() as session:
        stmt = (
            select(Task)
            .options(selectinload(Task.user))
            .options(selectinload(Task.logs))
            .where(Task.id == task_id)
        )
        result = await session.execute(stmt)
        return result.scalar_one_or_none()

async def 批量更新任务状态(task_ids: List[int], status: str):
    async with AsyncSessionLocal() as session:

            update(Task)
            .where(Task.id.in_(task_ids))
            .values(status=status)
        )
```
